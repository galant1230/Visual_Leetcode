class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        result = []

        def InOrder(node):
            if node:
                InOrder(node.left)
                result.append(node.val)
                InOrder(node.right)

        InOrder(root)

        for i in range(1, len(result)):
            if result[i] <= result[i - 1]:
                return False
        return True
