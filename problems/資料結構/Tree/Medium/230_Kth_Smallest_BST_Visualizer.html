class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        inorder_result = []
        def InOrder(root):
            if not root:
                return
            InOrder(root.left)
            inorder_result.append(root.val)
            InOrder(root.right)
        InOrder(root)
        return inorder_result[k - 1]
