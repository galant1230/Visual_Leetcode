queue = []
for i in range(numCourses):
    if indegree[i] == 0:
        queue.append(i)
while queue:
    node = queue.pop(0)
    visited += 1
    for neighbor in adj[node]:
        indegree[neighbor] -= 1
        if indegree[neighbor] == 0:
            queue.append(neighbor)
