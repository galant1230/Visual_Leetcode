import collections
class Solution:
    def minWindow(self, s, t):
        freq = dict(collections.Counter(t))
        count = 0
        slow = 0
        min_str = ""
        min_len = float("inf")
        for fast in range(len(s)):
            c = s[fast]
            if c in freq:
                freq[c] -= 1
                if freq[c] == 0:
                    count += 1
            while all(v <= 0 for v in freq.values()):
                if (fast - slow + 1) < min_len:
                    min_len = fast - slow + 1
                    min_str = s[slow:fast+1]
                left_c = s[slow]
                slow += 1
                if left_c in freq:
                    freq[left_c] += 1
                    if freq[left_c] > 0:
                        count -= 1
        return min_str
